function buildDailyRevenue() {
  // Use per-location per-day revenue from RAW.daily.rev
  const allDays = RAW.daily.days;
  const days = activeDay === 'all' ? allDays : [activeDay];
  const dayIndices = activeDay === 'all' ? [0,1,2] : [allDays.indexOf(activeDay)];

  let revData;
  if (activeLocation === 'all') {
    revData = dayIndices.map(i => RAW.daily.totalRev[i]);
  } else {
    revData = dayIndices.map(i => RAW.daily.rev[activeLocation][i]);
  }

  const barColors = ['#3B82F6','#60A5FA','#93C5FD'];
  const colors = dayIndices.map(i => barColors[i]);

  const trace = {
    x: days,
    y: revData,
    type: 'bar',
    marker: { color: colors, line:{width:0} },
    text: revData.map(v => `$${Math.round(v).toLocaleString('en-US')}`),
    textposition: 'outside',
    textfont: { color: '#E8EEF7', size: 12, family:"'DM Sans',sans-serif" },
    hovertemplate: '<b>%{x}</b><br>Revenue: $%{y:,.2f}<extra></extra>',
  };

  Plotly.newPlot('chart-daily', [trace], {
    ...DARK_LAYOUT,
    margin: { t:20, b:40, l:60, r:40 },
    xaxis: { ...DARK_LAYOUT.xaxis, type:'category' },
    yaxis: { ...DARK_LAYOUT.yaxis, tickprefix:'$', tickformat:',.0f' },
    height: 240,
    bargap: 0.35,
  }, CONFIG);
}

// ─── KPI UPDATE ────────────────────────────────────────────────────
function updateKPIs() {
  const dayFactors = {'all':1, 'Nov 26': 5515.65/8731.47, 'Nov 27': 1600.92/8731.47, 'Nov 28': 1614.90/8731.47};
  const totalRevAll = 8731.47; // sum of location subtotals (1268.84+3582.82+3879.81)
  const df = dayFactors[activeDay];

  let rev, orders, items;
  if (activeLocation === 'all') {
    rev = totalRevAll * df;
    orders = 27 * df;
    items = 54 * df;
  } else {
    rev = LOC_TOTALS[activeLocation] * df;
    orders = LOC_ORDERS[activeLocation] * df;
    items = Math.round(54 * (LOC_TOTALS[activeLocation] / totalRevAll) * df);
  }

  document.getElementById('kpi-revenue').textContent = '$' + rev.toLocaleString('en-US',{maximumFractionDigits:2,minimumFractionDigits:2});
  document.getElementById('kpi-orders').textContent = Math.round(orders);
  document.getElementById('kpi-aov').textContent = orders > 0 ? '$' + (rev/Math.round(orders)).toLocaleString('en-US',{maximumFractionDigits:2,minimumFractionDigits:2}) : '-';
  document.getElementById('kpi-items').textContent = Math.round(items);
}

    if (['all','Mammoth','Stowe','Sun Valley'].includes(b.textContent.replace(/[^a-zA-Z ]/g,'').trim()) ||
        b.onclick?.toString().includes('filterLocation')) {
      b.classList.remove('active');
    }
  });
  el.classList.add('active');
  renderAll();
}

function filterDay(day, el) {
  activeDay = day;
  // Only toggle day buttons
  const dayBtns = [...document.querySelectorAll('.filter-btn')].filter(b => 
    b.onclick?.toString().includes('filterDay'));
  dayBtns.forEach(b => b.classList.remove('active'));
  el.classList.add('active');
  renderAll();
}
